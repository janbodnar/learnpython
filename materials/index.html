apples.py
#!/usr/bin/python3

# number of baskets
baskets = 16

# number of apples in a basket
apples_in_basket = 24

# we get the total number of apples
total = baskets * apples_in_basket

print("There are total of", total, "apples")
fahrenheit.py
#!/usr/bin/python3

# fahrenheit.py

def C2F(c):
return c * 9/5 + 32

print(C2F(100))
print(C2F(0))
print(C2F(30))
